Basic Example #6

            $(document).ready(function() {
                $('#sample').ratchet(function() {
                    this.get("/", function(el) {
                        el.html("Click a page")
                        .append("<br/>")
                        .append("<a href='#/page1'>Page1</a>")
                        .append("<br/>")
                        .append("<a href='#/page2'>Page2</a>");
                        .append("<br/>")
                        .append("<a href='#/a/longer/url'>a/longer/url</a>")
                        .append("<br/>")
                        .append("<a href='#/something/else/that/we/want/to/invoke'>/something/else/that/we/want/to/invoke</a>");
                        // swap our content in
                        el.swap();
                    });

                    this.get("/**", function(el) {
                        el.html("You clicked: " + el.tokens['**']).swap();
                    });

                }).run("/");
            });
