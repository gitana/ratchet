Basic Example #7

            $(document).ready(function() {
                $('#sample').ratchet(function() {
                    this.get("/", function(el) {
                        var clickable = $("Click on me to pop up an alert");
                        clickable.click(function() {
                            alert("Here it is");
                        });

                        el.append(clickable);

                        el.swap();
                    });
                }).run("/");
            });
